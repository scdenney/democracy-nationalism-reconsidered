% preamble.tex — shared preamble for the JOP short-article submission (main + SI).
% Engine: pdflatex. EB Garamond text, 12pt, 1in margins. Double-spaced per JOP
% (text, footnotes, and references) and matching the ROC house style; the JOP
% short-article limit is 14 pages inclusive of text, notes, references, and floats.
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage[margin=1in]{geometry}
\usepackage{ebgaramond}                 % house serif
\usepackage{microtype}
\usepackage{setspace}
\usepackage{amsmath}
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{longtable}
\usepackage{tabularx}
\usepackage{array}
\usepackage{calc}                       % pandoc longtable column-width arithmetic
\providecommand{\real}[1]{#1}           % pandoc table macro (defined only in standalone mode)
\usepackage{float}
\usepackage{caption}
\usepackage{enumitem}
\usepackage{rotating}      % sidewaystable for wide SI tables
\usepackage[round]{natbib}
\bibliographystyle{apsr}   % JOP-appropriate author-date (APSR style)
\setlength{\bibsep}{2pt plus 0.5pt minus 0.5pt}   % tighten the reference list
\usepackage[hidelinks]{hyperref}
\usepackage[nameinlink,noabbrev]{cleveref}

% --- spacing: double-spaced body, footnotes, and references (JOP + ROC house style) ---
\doublespacing
% Keep tables compact (single-spaced) even where the surrounding body is double-spaced.
\usepackage{etoolbox}
\AtBeginEnvironment{longtable}{\singlespacing}
\AtBeginEnvironment{tabular}{\singlespacing}

% --- Unicode glyphs that appear in the prose/tables (pdflatex) ---
\usepackage{newunicodechar}
\newunicodechar{−}{\ensuremath{-}}        % U+2212 minus
\newunicodechar{→}{\ensuremath{\rightarrow}} % U+2192 arrow
\newunicodechar{–}{\textendash}
\newunicodechar{—}{\textemdash}

% --- table + caption defaults ---
% House style: bold label, colon separator, justified, no single-line centering.
\captionsetup{labelfont=bf, labelsep=colon, justification=justified, singlelinecheck=false, font=small}
% \figcap{title}{note}: the caption carries the title (small) with an italic "Note:" line
% beneath in a smaller font, no extra space before it, to keep captions compact.
\newcommand{\figcap}[2]{\caption[#1]{#1\\{\footnotesize\textit{Note:} #2}}}
\setlength{\tabcolsep}{4pt}
\renewcommand{\arraystretch}{1.05}
\newcolumntype{Y}{>{\raggedright\arraybackslash}X}
\providecommand{\tightlist}{}      % pandoc compatibility
\setlist{nosep}
\interfootnotelinepenalty=10000
\setlength{\emergencystretch}{3em}
\setlength{\skip\footins}{20pt plus 4pt minus 2pt}  % space between the text block and footnotes
\setlength{\footnotesep}{10pt}                       % space between successive footnotes

% pandoc image-bounding macro (scales a figure to fit the text block; pandoc defines
% this only in standalone mode, so we provide it for the converted bodies).
\makeatletter
\newsavebox\pandoc@box
\newcommand*\pandocbounded[1]{%
  \sbox\pandoc@box{#1}%
  \Gscale@div\@tempa\textheight{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
  \Gscale@div\@tempb\linewidth{\wd\pandoc@box}%
  \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi%
  \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
  \else\usebox\pandoc@box\fi}
\makeatother
